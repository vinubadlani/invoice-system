  --
  -- Name: sales_analytics; Type: VIEW; Schema: public; Owner: postgres
  --
  
  CREATE VIEW public.sales_analytics AS
   SELECT business_id,
      date_trunc('day'::text, (date)::timestamp with time zone) AS sale_date,
      date_trunc('week'::text, (date)::timestamp with time zone) AS sale_week,
      date_trunc('month'::text, (date)::timestamp with time zone) AS sale_month,
      count(*) AS invoice_count,
      sum(subtotal) AS total_subtotal,
      sum(total_tax) AS total_gst,
      sum(net_total) AS total_amount,
      sum(payment_received) AS total_received,
      sum(balance_due) AS total_pending,
      avg(net_total) AS average_order_value
     FROM public.invoices i
    WHERE (type = 'sales'::text)
    GROUP BY business_id, (date_trunc('day'::text, (date)::timestamp with time zone)), (date_trunc('week'::text, (date)::timestamp with time zone)), (date_trunc('month'::text, (date)::timestamp with time zone));
  
  
  ALTER VIEW public.sales_analytics OWNER TO postgres;
